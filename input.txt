import ioPlus;

class IfWhileNested {
    public int func(){
        boolean flag;

        flag = true;
        flag = !flag;

        if(flag){
            ioPlus.printResult(1);
        } else {
            ioPlus.printResult(2);
        }

        flag = !flag;
        return 1;
    }

    public static void main(String[] args) {
        IfWhileNested d;
        int a;
        d = new IfWhileNested();
        a = d.func();
    }
}
